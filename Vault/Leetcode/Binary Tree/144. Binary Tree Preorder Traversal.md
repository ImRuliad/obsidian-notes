class Solution:

def dfs(self, node: Optional[TreeNode], out: List[int]) -> None:

if not node:

return

out.append(node.val)

self.dfs(node.left, out)

self.dfs(node.right, out)

  

def preorderTraversal(self, root: Optional[TreeNode]) -> List[int]:

res = []

self.dfs(root, res)

return res
