import heapq
class Solution:
	def lastStoneWeight(self, stones: List[int]) -> int:
		h = []
		for weight in stones:
			heapq.heappush(h, -weight)
			
		while len(h) > 1:
			stone_1 = heapq.heappop(h)
			stone_2 = heapq.heappop(h)
			if stone_1 != stone_2:
				new_weight = stone_1 - stone_2
				heapq.heappush(h, new_weight)
		return -h[0] if h else 0
